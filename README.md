# story
# 登場人物と要素を定義
hero = "カナ"
goal = "AI教育の世界を変えること"
obstacle = "自信が持てないこと"
mentor = "未来から来たAI教師"
tool = "Pythonで動く魔法のノート"

# ストーリー展開を関数で表現
def tell_story(hero, goal, obstacle, mentor, tool):
    print(f"{hero}は、{goal}という夢を持っていた。")
    print(f"しかし、{obstacle}という壁にぶつかってしまう。")
    print(f"そこに、{mentor}が現れて、{tool}を授けてくれた。")
    print(f"{hero}は努力を重ねて、ついに{goal}を叶えた。")
    print("そして、彼女は次の世代に知恵を伝えていった…。")

# 実行
tell_story(hero, goal, obstacle, mentor, tool)
